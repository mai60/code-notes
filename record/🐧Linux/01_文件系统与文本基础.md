## ⚙️ Practice
```bash
#1.查看当前路径
pwd

#2.创建并进入练习目录
mkdir linux_test
cd linux_test

#3.创建多级目录+空文件
mkdir -p data/log/app
mkdir data/log/app/ko

#4.查看目录、删除空目录、退回上级
ls -alh data/log/app
cd ./data/log/app
remdir ko
cd ../../..

#5.批量创建4个空文件
touch user.txt run.log mark.txt test.txt

#6.复制备份文件+创建收纳文件夹
cp user.txt user_bak.txt
mkdir file_store

#7.移动文件到文件夹
mv user_bak.txt mark.txt file_store/

#8.创建软连接
ln -s run.log log_link

#9.给mark.txt写入测试内容
cat <<EOF | tee ./file_store/mark.txt 
72
88
65
88
91
72
88
59
EOF

#10.文件查看
cat -n mark.txt
head -n 4 mark.txt
tail -n 3 mark.txt
less mark.txt(#按q退出)
tail -f run.log

#11.文本统计、排序、去重
wc -l mark.txt
wc -w mark.txt
wc -c mark.txt
sort -n mark.txt
sort -nr mark.txt
sort -n mark.txt | uniq
sort -n mark.txt | uniq -c

#12.清理文件
rm log.link
rm -rf file_store

#13.返回用户家目录
cd ~
